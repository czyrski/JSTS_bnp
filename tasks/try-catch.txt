3. W bloku try, wywołaj funkcję simulateClick(TARGET_SELECTOR).

5. Użyj bloku finally, aby wypisać komunikat "Action finished. Continuing test execution."



// Helper function to simulate a browser action that might fail (Must be provided by instructor)
function simulateClick(selector: string): void {
    if (selector === "#missing-btn") {
        throw new Error(`Element with selector ${selector} was not found.`);
    }
    console.log(`Element clicked successfully: ${selector}`);
}
